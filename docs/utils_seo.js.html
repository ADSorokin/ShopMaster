/**
 * Утилита для генерации SEO-данных (title, description, keywords) в зависимости от страницы.
 *
 * @module seo
 *
 * @example
 * const seo = getSeoData('electronics');
 * document.title = seo.title;
 *
 * @example
 * const seo = getSeoData('product', product);
 * updateMetaTags(seo);
 */

/**
 * Возвращает SEO-данные для текущей страницы или товара.
 *
 * @param {string} currentPage - Текущая страница ('home', 'electronics', 'clothing', 'accessories', 'product')
 * @param {Object|null} [currentProduct=null] - Объект товара (только для страницы товара)
 * @returns {Object} Объект с полями `title`, `description`, `keywords`
 * @property {string} title - Заголовок страницы (до 60 символов)
 * @property {string} description - Описание для поисковых систем (до 160 символов)
 * @property {string} keywords - Ключевые слова через запятую
 *
 * @example
 * getSeoData('home');
 * // → { title: "ShopMaster - Лучший интернет-магазин...", ... }
 *
 * @example
 * getSeoData('product', {
 *   name: { ru: 'Смартфон X1' },
 *   description: { ru: 'Флагманский смартфон...' },
 *   price: 49990,
 *   brand: 'TechPro',
 *   specifications: { ru: { partNumber: 'SMX1-2024' } }
 * });
 * // → { title: "Смартфон X1 - купить в интернет-магазине ShopMaster", ... }
 *
 * @description
 * Особенности:
 * - Для страниц категорий возвращает предопределённые SEO-тексты
 * - Для страницы товара генерирует динамические мета-данные
 * - Автоматически использует русские названия (для SEO)
 * - При неизвестной странице возвращает данные для главной
 */
export const getSeoData = (currentPage, currentProduct = null) => {
    // === Базовые SEO-данные по страницам ===
    const seoData = {
        home: {
            title: 'ShopMaster - Лучший интернет-магазин электроники и товаров для дома',
            description: 'Купите качественные товары по лучшим ценам. Широкий ассортимент электроники, одежды и аксессуаров с доставкой по всей России.',
            keywords: 'интернет магазин, электроника, смартфоны, ноутбуки, купить онлайн, доставка, скидки'
        },
        electronics: {
            title: 'Электроника - Широкий выбор смартфонов, ноутбуков и гаджетов',
            description: 'Лучшие смартфоны, ноутбуки, наушники и другие электронные устройства по выгодным ценам с гарантией качества.',
            keywords: 'электроника, смартфоны, ноутбуки, наушники, гаджеты, купить электронику'
        },
        clothing: {
            title: 'Одежда и обувь - Стильная одежда по доступным ценам',
            description: 'Модная одежда и обувь для мужчин и женщин. Качественные материалы, современные фасоны и лучшие бренды.',
            keywords: 'одежда, обувь, кроссовки, футболки, купить одежду, мода'
        },
        accessories: {
            title: 'Аксессуары - Стильные аксессуары для каждого',
            description: 'Широкий выбор аксессуаров: сумки, рюкзаки, украшения и другие товары для повседневного использования.',
            keywords: 'аксессуары, сумки, рюкзаки, украшения, стиль, мода'
        }
    };

    // === Динамический SEO для страницы товара ===
    if (currentPage === 'product' && currentProduct) {
        const productName = typeof currentProduct.name === 'object'
            ? currentProduct.name.ru
            : currentProduct.name;

        const productDescription = typeof currentProduct.description === 'object'
            ? currentProduct.description.ru
            : currentProduct.description;

        const partNumber = currentProduct.specifications?.ru?.partNumber || '';
        const price = currentProduct.price.toLocaleString('ru-RU');

        return {
            title: `${productName} - купить в интернет-магазине ShopMaster`,
            description: `${productDescription} по цене ${price} ₽ с доставкой по всей России.`,
            keywords: `${productName}, ${currentProduct.brand}, ${partNumber}, купить, цена, характеристики, отзывы`
        };
    }

    // === Резерв: главная страница ===
    return seoData[currentPage] || seoData.home;
};
