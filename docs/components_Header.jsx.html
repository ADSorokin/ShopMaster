                  {compareCount}

              )}
            </button>
            
            {/* Иконка корзины */}
            <button

                >
                  {getTranslation('logout')}
                </button>
              </div>

              <button
                onClick={() => setCurrentPage('login')}
                className="text-sm text-indigo-600 hover:text-indigo-800 font-medium"
              >
                {getTranslation('login')}
              </button>

          </div>
